"use client";
import { useState } from "react";

export default function Home() {
  const [loading, setLoading] = useState(false);
  const [status, setStatus] = useState("ready"); // ready, processing, success, error
  const [fileName, setFileName] = useState("");
  const [errorMessage, setErrorMessage] = useState("");

  const resetStatus = () => {
    setStatus("ready");
    setFileName("");
    setErrorMessage("");
    setLoading(false);
  };

  async function handleUpload(e) {
    const file = e.target.files[0];
    if (!file) return;

    setFileName(file.name);
    setLoading(true);
    setStatus("processing");
    setErrorMessage("");

    try {
      const formData = new FormData();
      formData.append("file", file);

      const res = await fetch("/api/sort", {
        method: "POST",
        body: formData,
      });

      // Extrae el mensaje de error personalizado enviado desde el backend de la aplicación
      if (!res.ok) {
        const errorData = await res.json();
        throw new Error(errorData.error || "Error interno en el servidor");
      }

      const blob = await res.blob();
      const url = window.URL.createObjectURL(blob);
      const a = document.createElement("a");
      a.href = url;
      a.download = `ORDENADO_${file.name}`;
      a.click();
      window.URL.revokeObjectURL(url);
      
      setStatus("success");
    } catch (err) {
      console.error(err);
      setErrorMessage(err.message);
      setStatus("error");
    } finally {
      setLoading(false);
    }
  }

  return (
    <main className="min-h-screen bg-[#f8fafc] flex flex-col items-center py-12 px-4 font-sans">
      {/* Header */}
      <div className="max-w-2xl w-full mb-8 text-center">
        <div className="inline-block bg-blue-100 text-blue-700 text-xs font-bold px-3 py-1 rounded-full mb-4 uppercase tracking-widest">
          TIKTOK SHOP - ALMACENAMIENTO DE GUÍAS PDF
        </div>
        <h1 className="text-4xl font-black text-red-900 mb-2 tracking-tight">
          J&T <span className="text-red-600">EXPRESS</span>
        </h1>
        <p className="text-slate-500 font-medium">
          Sistema de ordenamiento masivo por SKU agrupados
        </p>
      </div>

      {/* Card Principal */}
      <div className="max-w-xl w-full bg-white rounded-3xl shadow-2xl shadow-blue-100/50 overflow-hidden border border-slate-100">
        <div className="p-10">
          <div 
            className={`relative border-2 border-dashed rounded-2xl p-12 transition-all duration-300 flex flex-col items-center justify-center ${
              status === "processing" ? "border-blue-400 bg-blue-50/50 animate-pulse" : 
              status === "success" ? "border-green-400 bg-green-50/50" :
              status === "error" ? "border-red-400 bg-red-50/50" : "border-slate-200 hover:border-blue-400 hover:bg-slate-50"
            }`}
          >
            {/* Estado Visual */}
            <div className="mb-6 transform transition-transform duration-500 hover:scale-110">
              {status === "ready" && <div className="text-6xl text-slate-300">📤</div>}
              {status === "processing" && <div className="text-6xl animate-bounce">⚙️</div>}
              {status === "success" && <div className="text-6xl drop-shadow-md">🎉</div>}
              {status === "error" && <div className="text-6xl">⚠️</div>}
            </div>

            <div className="text-center w-full">
              <h3 className="text-xl font-bold text-slate-800 mb-2">
                {status === "ready" && "Cargar guías PDF"}
                {status === "processing" && "Organizando por SKU..."}
                {status === "success" && "¡Proceso Completado!"}
                {status === "error" && "Tienda o almacén inválido"}
              </h3>
              
              <div className="text-sm text-slate-500 mb-6 max-w-[320px] mx-auto">
                {status === "ready" && "Selecciona el archivo masivo descargado de TikTok Shop."}
                {status === "processing" && `Analizando contenido de: ${fileName}`}
                {status === "success" && "El archivo ordenado se ha descargado automáticamente."}
                {status === "error" && (
                  <p className="text-red-600 font-semibold bg-red-100/60 p-2.5 rounded-xl border border-red-200/50 mt-2">
                    {errorMessage}
                  </p>
                )}
              </div>

              {/* Botón Dinámico */}
              {status === "ready" ? (
                <label className="cursor-pointer inline-flex items-center justify-center px-8 py-3 bg-blue-600 text-white rounded-xl font-bold shadow-lg shadow-blue-200 hover:bg-blue-700 hover:-translate-y-0.5 active:translate-y-0 transition-all">
                  <span>Seleccionar Archivo</span>
                  <input type="file" accept=".pdf" className="hidden" onChange={handleUpload} />
                </label>
              ) : status === "success" ? (
                <button 
                  onClick={resetStatus}
                  className="inline-flex items-center justify-center px-8 py-3 bg-slate-800 text-white rounded-xl font-bold shadow-lg shadow-slate-200 hover:bg-slate-900 hover:-translate-y-0.5 active:translate-y-0 transition-all"
                >
                  🔄 Procesar otro archivo
                </button>
              ) : status === "error" ? (
                <button 
                  onClick={resetStatus}
                  className="inline-flex items-center justify-center px-8 py-3 bg-red-600 text-white rounded-xl font-bold shadow-lg shadow-red-200 hover:bg-red-700 transition-all"
                >
                  Reintentar
                </button>
              ) : null}
            </div>
          </div>

          {/* Badge de Seguridad */}
          <div className="mt-8 flex items-center justify-center gap-4 py-3 bg-slate-50 rounded-xl border border-slate-100">
            <div className="flex items-center gap-1.5">
              <span className="text-green-500 text-lg">🛡️</span>
              <span className="text-[10px] font-bold text-slate-400 uppercase tracking-wider text-center leading-tight">
                Protección de Datos <br/> Red Local Activa
              </span>
            </div>
            <div className="w-px h-6 bg-slate-200"></div>
            <div className="flex items-center gap-1.5">
              <span className="text-blue-500 text-lg">⚡</span>
              <span className="text-[10px] font-bold text-slate-400 uppercase tracking-wider text-center leading-tight">
                Procesamiento <br/> Ultra Rápido
              </span>
            </div>
          </div>
        </div>
        
        {/* Footer info */}
        <div className="bg-slate-900 px-8 py-4 flex justify-between items-center">
          <div className="flex items-center gap-2">
            <div className="w-2 h-2 bg-green-400 rounded-full animate-pulse"></div>
            <span className="text-white/70 text-[10px] font-bold uppercase tracking-widest">Servidor Local Online</span>
          </div>
          <span className="text-white/40 text-[10px] font-medium">Bolsa de soporte activa: 48 Hrs</span>
        </div>
      </div>

      <p className="mt-8 text-slate-400 text-[10px] font-bold uppercase tracking-[0.2em]">
        Desarrollado para Operaciones de Alto Volumen v1.0
      </p>
    </main>
  );
}
