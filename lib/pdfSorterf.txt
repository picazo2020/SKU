import { PDFDocument } from "pdf-lib";
import PdfParser from "pdf2json";

const getPdfText = (buffer) => {
  return new Promise((resolve, reject) => {
    const pdfParser = new PdfParser();
    pdfParser.on("pdfParser_dataReady", (pdfData) => {
      const pages = pdfData.Pages.map(page => {
        return page.Texts.map(t => {
          if (t.R && t.R[0] && t.R[0].T) {
            // 🎯 BLOQUE SEGURO: Si el texto viene malformado, evita que la app se truene
            try {
              return decodeURIComponent(t.R[0].T);
            } catch (error) {
              return t.R[0].T; // Conserva el texto original de respaldo
            }
          }
          return "";
        }).join(""); 
      });
      resolve(pages);
    });
    pdfParser.on("pdfParser_dataError", (err) => reject(err));
    pdfParser.parseBuffer(buffer);
  });
};

export async function sortPdfPairsBySku(buffers) {
  try {
    const finalPdf = await PDFDocument.create();
    const allPairsGlobal = [];
    const skuRegex = /(\d{4})[-]*TRR/i;

    // Iteramos sobre todos los archivos PDF seleccionados
    for (let b = 0; b < buffers.length; b++) {
      const buffer = buffers[b];
      const allPagesText = await getPdfText(buffer);
      const originalPdf = await PDFDocument.load(buffer);
      const totalPages = originalPdf.getPageCount();

      // Extraemos las hojas y el SKU de cada pedido de forma nativa
      for (let i = 0; i < totalPages; i += 2) {
        const p1 = i;
        const p2 = i + 1;
        
        const textToSearch = allPagesText[p2] || "";
        const match = textToSearch.match(skuRegex);
        const skuPrefix = (match && match[1]) ? match[1] : "9999";

        // Clonamos el par de páginas de este archivo individual
        const valid = [p1, p2].filter(idx => idx < totalPages);
        const copiedPages = await finalPdf.copyPages(originalPdf, valid);

        allPairsGlobal.push({
          sku: skuPrefix,
          pages: copiedPages
        });
      }
    }

    // Ordenamos la lista combinada global numéricamente de menor a mayor
    allPairsGlobal.sort((a, b) => parseInt(a.sku) - parseInt(b.sku));

    // Añadimos las páginas ordenadas al documento consolidado final
    for (const pair of allPairsGlobal) {
      pair.pages.forEach(page => finalPdf.addPage(page));
    }

    console.log(`\n✨ Proceso completado: Se mezclaron y ordenaron ${allPairsGlobal.length} pedidos totales.`);
    return await finalPdf.save();
  } catch (error) {
    console.error("Error crítico en ordenamiento múltiple:", error);
    throw error;
  }
}
